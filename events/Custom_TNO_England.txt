					set_variable = { TNO_Britain_ElectionSeason_AP_v = 25 }

		else_if = {
				limit = { has_game_rule = { rule = HMMLR_IDEOLOGY_ROUTE option = HMMLR_IDEOLOGY_ROUTE_LIBSOC } }
