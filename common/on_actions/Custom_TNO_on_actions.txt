					        tag = OST
					        tag = DST
					        tag = MST
					        tag = JST
					        tag = UPO
					        tag = BLR

					}

					        tag = MST
					        tag = JST
					        tag = UPO
					        tag = BLR
					    }
					}
				}
				if = {
				    limit = {
				        has_game_rule = {
						    rule = ostland_civil_war_outcome
						    option = OSTLAND_CIVIL_WAR_OUTCOME_OST_STAHLECKER
					    }
					    NOT = {
					        has_global_flag = OST_CIVIL_WAR_BUFF_APPLIED
					    }

				        has_game_rule = {
						    rule = ostland_civil_war_outcome
						    option = OSTLAND_CIVIL_WAR_OUTCOME_UPO_KOVNER
					    }

					        has_global_flag = OST_CIVIL_WAR_BUFF_APPLIED
					    }
				    }

					    }
				    }
				    set_global_flag = OST_CIVIL_WAR_BUFF_APPLIED
				    BLR = { apply_war_buff_with_10_divisions = yes }				    
				}
			}

	}

					        tag = OST

					        tag = MST
					        tag = JST
					        tag = UPO
					        tag = BLR
					    }
					}

					    OR = {
					        tag = OST
					        tag = DST
					        tag = MST
					        tag = JST
					        tag = UPO
					        tag = BLR
					    }
					}
					NOT = {
			            count_triggers = {
				            amount = 2 
				            country_exists = OST
				            country_exists = DST
				            country_exists = MST
				            country_exists = JST
				            country_exists = UPO
				            country_exists = BLR
			            }
		            }
				}
				if = {
				    limit = {

						    rule = ostland_civil_war_outcome
