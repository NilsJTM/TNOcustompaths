	set_country_flag = act_progress #acts foci shouldn't trigger while this flag is active
	clear_array = ActMechanic_MPs #just for safety; this should already have been cleared

	multiply_temp_variable = { UE_Liberal_Support = ParliamentMechanic_Party_SeatNum^6 }
	add_to_array = { ActMechanic_MPs = UE_Liberal_Support }

	multiply_temp_variable = { RP_Support = ParliamentMechanic_Party_SeatNum^7 }
	add_to_array = { ActMechanic_MPs = RP_Support }

	multiply_temp_variable = { NF_Support = ParliamentMechanic_Party_SeatNum^8 }
	add_to_array = { ActMechanic_MPs = NF_Support }

	multiply_temp_variable = { Liberal_Support = ParliamentMechanic_Party_SeatNum^9 }
	add_to_array = { ActMechanic_MPs = Liberal_Support }

	multiply_temp_variable = { NDL_Whig_Support = ParliamentMechanic_Party_SeatNum^10 }
	add_to_array = { ActMechanic_MPs = NDL_Whig_Support }

	multiply_temp_variable = { NDL_Democrat_Support = ParliamentMechanic_Party_SeatNum^11 }
	add_to_array = { ActMechanic_MPs = NDL_Democrat_Support }

	multiply_temp_variable = { NDL_Patriot_Support = ParliamentMechanic_Party_SeatNum^12 }
	add_to_array = { ActMechanic_MPs = NDL_Patriot_Support }

	multiply_temp_variable = { SLP_Reform_Support = ParliamentMechanic_Party_SeatNum^13 }
	add_to_array = { ActMechanic_MPs = SLP_Reform_Support }

	multiply_temp_variable = { SLP_Hardliner_Support = ParliamentMechanic_Party_SeatNum^14 }
	add_to_array = { ActMechanic_MPs = SLP_Hardliner_Support }

	multiply_temp_variable = { Movement_Support = ParliamentMechanic_Party_SeatNum^15 }
	add_to_array = { ActMechanic_MPs = Movement_Support }

	multiply_temp_variable = { ILP_Support = ParliamentMechanic_Party_SeatNum^16 }
	add_to_array = { ActMechanic_MPs = ILP_Support }

	multiply_temp_variable = { SNP_Support = ParliamentMechanic_Party_SeatNum^17 }
	add_to_array = { ActMechanic_MPs = SNP_Support }

	multiply_temp_variable = { UP_Support = ParliamentMechanic_Party_SeatNum^18 }
	add_to_array = { ActMechanic_MPs = UP_Support }

	multiply_temp_variable = { Plaid_Support = ParliamentMechanic_Party_SeatNum^19 }
	add_to_array = { ActMechanic_MPs = Plaid_Support }

	multiply_temp_variable = { Goch_Support = ParliamentMechanic_Party_SeatNum^20 }
	add_to_array = { ActMechanic_MPs = Goch_Support }

	multiply_temp_variable = { BNC_Support = ParliamentMechanic_Party_SeatNum^21 }
	add_to_array = { ActMechanic_MPs = BNC_Support }

	multiply_temp_variable = { PC_Support = ParliamentMechanic_Party_SeatNum^22 }
	add_to_array = { ActMechanic_MPs = PC_Support }

	for_each_loop = {
		array = ParliamentMechanic_Party_SeatNum
		add_to_variable = { totalseats = v }
	}

	set_variable = { half_totalseats = totalseats }
	add_to_variable = { half_totalseats = 1 }
	multiply_variable = { half_totalseats = 0.5 }
	round_variable = half_totalseats

	for_each_loop = {
		array = ActMechanic_MPs
		randomize_temp_variable = {
			var = Act_Randomization
			distribution = binomial
			min = .90
			max = 1
		}
		multiply_variable = { ActMechanic_MPs^i = Act_Randomization }
		round_variable = ActMechanic_MPs^i
	}
	set_variable = { MPs_Support = 0 }
	for_each_loop = {
		array = ActMechanic_MPs
		add_to_variable = { MPs_Support = v }
	}

	hidden_effect = {
		activate_mission = ENG_Acts_Progress
	}

}

ENG_End_Act = {
	clear_variable = MPs_Support
	for_each_loop = {
		array = ActMechanic_MPs
		add_to_variable = { MPs_Support = v }
	}
	multiply_variable = { totalseats = 0.5 }
	if = {
		limit = {
			OR = {
				has_country_flag = ENG_instant_pass
				check_variable = { MPs_Support > totalseats }
				has_game_rule = {
					rule = ENG_PASS_ACT_CHEAT
					option = ENG_PASS_ACT_CHEAT_YES
				} 
			}
			NOT = { has_country_flag = ENG_instant_fail }
		}
		country_event = ENG_acts.1
		else = {
			country_event = ENG_acts.2
		}
	}

	clear_array = ActMechanic_MPs
	clear_variable = MPs_Support
	clear_variable = totalseats
}
